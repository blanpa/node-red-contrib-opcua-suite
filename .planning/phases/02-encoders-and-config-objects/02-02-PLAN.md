DataSetMessage model:
```js
{
  dataSetWriterId,              // UInt16 — repeated also in PayloadHeader; kept in model for symmetry
  fieldEncoding,                // 'variant' | 'datavalue' | 'rawdata'  (default 'variant')
  messageType,                  // 'keyframe' | 'deltaframe' | 'keepalive' | 'event'  (default 'keyframe')
  valid,                        // boolean — default true (DataSetFlags1 bit 0)
  sequenceNumber,               // UInt16 — sets DataSetFlags1 bit 3
  status,                       // UInt16 StatusCode short form — sets bit 4
  configurationVersion,         // { major: UInt32, minor: UInt32 } — sets bits 5, 6
  timestamp,                    // JS Date — sets DataSetFlags2 bit 4
  picoseconds,                  // UInt16 — sets DataSetFlags2 bit 5 (requires timestamp)
  fields                        // For variant: { [name]: { dataType, value } }
                                // For datavalue: { [name]: { value, statusCode?, sourceTimestamp?, serverTimestamp? } }
                                // For rawdata: { [name]: <typed value> } — ordered, type info comes from external metadata
}
```

DataSetFlags1 (Part 14 §7.2.4.5.4, VERIFIED 2026-05-13):
- bit 0 (0x01): Valid
- bits 1-2: FieldEncoding (00=Variant, 01=RawData, 10=DataValue, 11=reserved)
- bit 3 (0x08): SequenceNumber enabled
- bit 4 (0x10): Status enabled
- bit 5 (0x20): ConfigVersionMajor enabled
- bit 6 (0x40): ConfigVersionMinor enabled
- bit 7 (0x80): DataSetFlags2 enabled

DataSetFlags2:
- bits 0-3: Message Type (0000=KeyFrame, 0001=DeltaFrame, 0010=Event, 0011=KeepAlive)
- bit 4 (0x10): Timestamp enabled
- bit 5 (0x20): PicoSeconds enabled
- bits 6-7: Reserved (0)

Suppression rule: DataSetFlags2 SHALL be omitted when all bits = 0 (default KeyFrame, no per-message timestamp); DataSetFlags1 bit 7 MUST be cleared in that case.

Variant binary encoding (Part 6 §5.2.2.16, simplified for v1 — only common scalar types):
```
[EncodingByte: UInt8]
  bits 0-5: BuiltInType id
  bit 6 (0x40): IsArray (1 = array follows)
  bit 7 (0x80): HasDimensions (1 = dimensions array follows)
[Value: type-specific bytes — for v1 support: Boolean(1), SByte(2), Byte(3), Int16(4), UInt16(5), Int32(6), UInt32(7), Int64(8), UInt64(9), Float(10), Double(11), String(12), DateTime(13), Guid(14), ByteString(15), StatusCode(19)]
```

DataValue binary encoding (Part 6 §5.2.2.17):
```
[EncodingMask: UInt8]
  bit 0: Value (Variant) present
  bit 1: Status present (UInt32)
  bit 2: SourceTimestamp present (DateTime)
  bit 3: ServerTimestamp present (DateTime)
  bit 4: SourcePicoseconds present (UInt16)
  bit 5: ServerPicoseconds present (UInt16)
[Value: Variant if bit 0]
[Status: UInt32 if bit 1]
[SourceTimestamp: DateTime if bit 2]
[ServerTimestamp: DateTime if bit 3]
[SourcePicoseconds: UInt16 if bit 4]
[ServerPicoseconds: UInt16 if bit 5]
```

Chunk wire layout (Part 14 §7.2.4.4.4):
- ExtendedFlags2 bit 0 = 1 marks a chunk NetworkMessage
- PayloadHeader has dataSetWriterIds with exactly one entry (the chunked DSW)
- Chunk payload (replaces normal DataSetMessage body):
  - MessageSequenceNumber: UInt16 (sequence number of the reassembled payload)
  - ChunkOffset: UInt32 (byte position within full payload)
  - TotalSize: UInt32
  - ChunkData: ByteString (UInt32 length + bytes)

function _readVariantScalar(stream, builtIn) {
  switch (builtIn) {
    case BUILTIN_TYPE.Boolean:    return stream.readUInt8() !== 0;
    case BUILTIN_TYPE.SByte:      { stream._ensureRead(1); const v = stream._buf.readInt8(stream._cursor); stream._cursor += 1; return v; }
    case BUILTIN_TYPE.Byte:       return stream.readUInt8();
    case BUILTIN_TYPE.Int16:      { stream._ensureRead(2); const v = stream._buf.readInt16LE(stream._cursor); stream._cursor += 2; return v; }
    case BUILTIN_TYPE.UInt16:     return stream.readUInt16LE();
    case BUILTIN_TYPE.Int32:      { stream._ensureRead(4); const v = stream._buf.readInt32LE(stream._cursor); stream._cursor += 4; return v; }
    case BUILTIN_TYPE.UInt32:     return stream.readUInt32LE();
    case BUILTIN_TYPE.Int64:      { stream._ensureRead(8); const v = stream._buf.readBigInt64LE(stream._cursor); stream._cursor += 8; return v; }
    case BUILTIN_TYPE.UInt64:     return stream.readUInt64LE();
    case BUILTIN_TYPE.Float:      { stream._ensureRead(4); const v = stream._buf.readFloatLE(stream._cursor); stream._cursor += 4; return v; }
    case BUILTIN_TYPE.Double:     { stream._ensureRead(8); const v = stream._buf.readDoubleLE(stream._cursor); stream._cursor += 8; return v; }
    case BUILTIN_TYPE.String:     return stream.readString();
    case BUILTIN_TYPE.DateTime:   return stream.readDateTime();
    case BUILTIN_TYPE.Guid:       return stream.readGuid();
    case BUILTIN_TYPE.ByteString: {
      const len = stream.readUInt32LE();

      return Buffer.from(stream.readBytes(len)); // copy out so caller owns it

function _writeVariant(stream, variant) {
  if (variant === null || variant === undefined) {
    stream.writeUInt8(0x00); // empty Variant
    return;
  }
  const { dataType, value } = variant;
  const builtIn = (typeof dataType === "number") ? dataType : BUILTIN_TYPE[dataType];
  if (!builtIn) throw createError(`UADP_VARIANT_UNKNOWN_TYPE: ${dataType}`);
  const isArray = Array.isArray(value);
  let encodingByte = builtIn & 0x3F;
  if (isArray) encodingByte |= 0x40;
  stream.writeUInt8(encodingByte);
  if (isArray) {
    stream._ensureWrite(4); stream._buf.writeInt32LE(value.length, stream._cursor); stream._cursor += 4;
    for (const v of value) _writeVariantScalar(stream, builtIn, v);
  } else {
    _writeVariantScalar(stream, builtIn, value);
  }
}

function _readVariant(stream) {
  const encodingByte = stream.readUInt8();
  if (encodingByte === 0x00) return null; // empty Variant
  const builtIn = encodingByte & 0x3F;
  const isArray = (encodingByte & 0x40) !== 0;
  const hasDims = (encodingByte & 0x80) !== 0;
  let value;
  if (isArray) {
    stream._ensureRead(4); const len = stream._buf.readInt32LE(stream._cursor); stream._cursor += 4;
    value = [];
    for (let i = 0; i < len; i++) value.push(_readVariantScalar(stream, builtIn));
  } else {
    value = _readVariantScalar(stream, builtIn);
  }
  if (hasDims) {
    stream._ensureRead(4); const dimCount = stream._buf.readInt32LE(stream._cursor); stream._cursor += 4;
    const dims = [];
    for (let i = 0; i < dimCount; i++) {
      stream._ensureRead(4); dims.push(stream._buf.readInt32LE(stream._cursor)); stream._cursor += 4;
    }
    return { dataType: builtIn, value, dimensions: dims };
  }
  return { dataType: builtIn, value };
}

function encodeDataSetMessage(dsm, _opts) {
  if (!dsm || typeof dsm !== "object") {
    throw createError("UADP_ENCODE_INVALID_INPUT: dataSetMessage must be an object");
  }
  const fieldEncoding = dsm.fieldEncoding || "variant";
  const messageType   = dsm.messageType   || "keyframe";
  const encodingBits  = FIELD_ENCODING_BITS[fieldEncoding];
  const messageBits   = MESSAGE_TYPE_BITS[messageType];
  if (encodingBits === undefined) throw createError(`UADP_ENCODE_INVALID_FIELD_ENCODING: ${fieldEncoding}`);
  if (messageBits === undefined)  throw createError(`UADP_ENCODE_INVALID_MESSAGE_TYPE: ${messageType}`);

  // Build DataSetFlags2 first (downstream cascade)
  let dsFlags2 = messageBits & 0x0F; // bits 0-3
  if (dsm.timestamp)                                 dsFlags2 |= 0x10; // bit 4
  if (dsm.picoseconds !== undefined && dsm.timestamp) dsFlags2 |= 0x20; // bit 5

  // Build DataSetFlags1
  let dsFlags1 = (dsm.valid !== false) ? 0x01 : 0x00; // bit 0
  dsFlags1 |= (encodingBits & 0x03) << 1;             // bits 1-2
  if (dsm.sequenceNumber !== undefined)        dsFlags1 |= 0x08; // bit 3
  if (dsm.status !== undefined)                dsFlags1 |= 0x10; // bit 4
  if (dsm.configurationVersion?.major !== undefined) dsFlags1 |= 0x20; // bit 5
  if (dsm.configurationVersion?.minor !== undefined) dsFlags1 |= 0x40; // bit 6
  if (dsFlags2 !== 0x00)                       dsFlags1 |= 0x80; // bit 7

  const stream = new BinaryStream(256);
  stream.writeUInt8(dsFlags1);
  if (dsFlags1 & 0x80) stream.writeUInt8(dsFlags2);

  if (dsFlags1 & 0x08) stream.writeUInt16LE(dsm.sequenceNumber);
  if (dsFlags1 & 0x10) stream.writeUInt16LE(dsm.status);
  if (dsFlags1 & 0x20) stream.writeUInt32LE(dsm.configurationVersion.major);
  if (dsFlags1 & 0x40) stream.writeUInt32LE(dsm.configurationVersion.minor);
  if (dsFlags2 & 0x10) stream.writeDateTime(dsm.timestamp);
  if (dsFlags2 & 0x20) stream.writeUInt16LE(dsm.picoseconds);

  // Field payload — only for keyframe/deltaframe/event (keepalive has no body)
  if (messageType !== "keepalive") {
    const names = Object.keys(dsm.fields || {});
    stream.writeUInt16LE(names.length);
    for (const name of names) {
      const nameBytes = Buffer.from(name, "utf8");
      stream.writeUInt16LE(nameBytes.length);
      stream.writeBytes(nameBytes);

      else if (fieldEncoding === "datavalue") _writeDataValue(stream, field);

        _writeVariantScalar(stream, builtIn, field.value);
      }
    }
  }
  return stream.toBuffer();
}

function _isChunkRequired(buffer, mtu) {
  return buffer.length > mtu;
}

function _splitIntoChunks(fullPayload, mtu, headerOverhead) {
  const maxChunkData = mtu - headerOverhead;
  const chunks = [];
  let offset = 0;
  while (offset < fullPayload.length) {
    const size = Math.min(maxChunkData, fullPayload.length - offset);
    chunks.push({
      chunkOffset: offset,
      totalSize:   fullPayload.length,
      chunkData:   fullPayload.subarray(offset, offset + size),
    });
    offset += size;
  }
  return chunks;
}

function _encodeChunkNetworkMessage(originalNm, chunkInfo, messageSequenceNumber, dataSetWriterId) {

  // and payload replaced by the chunk-payload struct.

    publisherId: originalNm.publisherId,

    payload: [],      // body is the chunk-payload struct, not a DataSetMessage

  // For implementation simplicity: encode the standard header, then append chunk fields directly.

  const extFlags2 = _buildExtendedFlags2(chunkNm);
  const extFlags1 = _buildExtendedFlags1(chunkNm, extFlags2);
  const uadpFlags = _buildUADPFlags(chunkNm, extFlags1);

  const stream = new BinaryStream(mtuEstimate(chunkInfo));
  stream.writeUInt8(uadpFlags);
  if (uadpFlags & 0x80) stream.writeUInt8(extFlags1);
  if ((uadpFlags & 0x80) && (extFlags1 & 0x80)) stream.writeUInt8(extFlags2);
  // Header writes (PublisherId, GroupHeader, PayloadHeader, Timestamp...) — reuse helper extracted from encodeNetworkMessage
  _writeNetworkMessageHeaderFields(stream, chunkNm, uadpFlags, extFlags1, extFlags2);
  // Chunk payload (Part 14 §7.2.4.4.4)
  stream.writeUInt16LE(messageSequenceNumber);
  stream.writeUInt32LE(chunkInfo.chunkOffset);
  stream.writeUInt32LE(chunkInfo.totalSize);
  stream.writeUInt32LE(chunkInfo.chunkData.length);
  stream.writeBytes(chunkInfo.chunkData);
  return stream.toBuffer();
}

function mtuEstimate(chunkInfo) {
  return chunkInfo.chunkData.length + 64; // 64-byte slack for headers
}
