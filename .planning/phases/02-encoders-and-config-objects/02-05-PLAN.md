function _populateNullHex(entry) {

    // Static-chunk fixtures rely on encoder support for model.chunk — currently throws.

  return entry;

"use strict";

const { expect } = require("chai");

const u = require("../lib/uadp-encoder");
const vectors = require("./fixtures/uadp-vectors");

// ─── Local Helpers ───

function hexStrip(s) { return s.replace(/[\s_]/g, "").toLowerCase(); }
function bufHex(buf) { return buf.toString("hex").toLowerCase(); }

describe("uadp-encoder", function () {

  // ─── module exports ───

  describe("module exports", function () {
    it("exports the four expected functions", function () {
      expect(u.encodeNetworkMessage).to.be.a("function");
      expect(u.decodeNetworkMessage).to.be.a("function");
      expect(u.encodeDataSetMessage).to.be.a("function");
      expect(u.decodeDataSetMessage).to.be.a("function");
    });
    it("does NOT export BinaryStream (private)", function () {
      expect(u.BinaryStream).to.equal(undefined);
    });
  });

  // ─── 8-combination flag cascade matrix (D-17, D-19, ENC-01 success criterion #1) ───

  describe("8-combination flag cascade matrix", function () {
    it("fixture file exports exactly 8 cases", function () {
      expect(Object.keys(vectors)).to.have.length(8);
    });

    for (const [name, vec] of Object.entries(vectors)) {
      describe(name, function () {
        it("first byte (UADPFlags) matches fixture.flags.uadpFlags", function () {
          if (vec.pending) this.skip();
          const encoded = u.encodeNetworkMessage(vec.model);
          // Chunked output is Array<Buffer> when MTU exceeded; for fixture cases the model is small so always Buffer
          const buf = Array.isArray(encoded) ? encoded[0] : encoded;
          expect(buf[0]).to.equal(vec.flags.uadpFlags);
        });

        it("encodeNetworkMessage(model) hex matches fixture.hex", function () {
          if (vec.pending) this.skip();
          if (vec.isStaticChunk) this.skip(); // static-chunk fixtures depend on encoder honoring model.chunk; tested separately
          const encoded = u.encodeNetworkMessage(vec.model);
          const buf = Buffer.isBuffer(encoded) ? encoded : encoded[0];
          expect(bufHex(buf)).to.equal(hexStrip(vec.hex));
        });

        it("decodeNetworkMessage(hex) round-trips to fixture.model", function () {
          if (vec.pending) this.skip();
          if (vec.isStaticChunk) this.skip();
          const buf = Buffer.from(hexStrip(vec.hex), "hex");
          const decoded = u.decodeNetworkMessage(buf);
          // Compare PublisherId presence and key fields. Use loose comparison: undefined fields in
          // decoder output are not present in fixture.model and vice versa.
          if (vec.model.publisherId !== undefined) {
            expect(decoded.publisherId).to.deep.equal(vec.model.publisherId);

      expect(() => u.decodeNetworkMessage(Buffer.from([0x02]))).to.throw(/UADP_DECODE_UNSUPPORTED_VERSION/);
    });

/**
 * UADP Reference Vector Capture Script
 *
 * Captures UADP binary packets from a running open62541 publisher and dumps
 * hex to stdout for pasting into test/fixtures/uadp-vectors.js.
 *
 * NOT picked up by `npm test` (does not end in .test.js — D-18).
 *
 * Setup:
 *   1. Pull the open62541 reference container (e.g., via Docker):
 *        docker pull open62541/open62541-publisher
 *   2. Start it with UADP publishing enabled and target port 4840:
 *        docker run --rm --network=host open62541/open62541-publisher \
 *          --pubsub-uadp --port=4840
 *   3. Run this script in a separate terminal:
 *        node test-server/capture-open62541-vectors.js
 *   4. Each captured packet prints as:
 *        // From <ip>:<port>
 *        <space-separated hex bytes>
 *   5. Copy the hex strings into test/fixtures/uadp-vectors.js, replacing the
 *      `hex: null` (encoder-self-output) entries and updating provenance to
 *      "open62541 v<version> captured <date>".
 *   6. Stop the script with Ctrl-C; the socket closes cleanly.
 *
 * Phase 4 will use this script to upgrade fixture provenance from
 * "encoder self-output" to verified open62541 reference output.
 *
 * @author node-red-contrib-opcua-suite
 * @see Part 14 §7.2.4 (UADP NetworkMessage encoding)
 */

const dgram = require("dgram");

const DEFAULT_PORT = 4840;

function formatHex(buf) {
  return buf.toString("hex").match(/.{2}/g).join(" ");
}

function main() {
  const port = Number(process.env.UADP_CAPTURE_PORT) || DEFAULT_PORT;
  const sock = dgram.createSocket({ type: "udp4", reuseAddr: true });

  sock.on("message", (msg, rinfo) => {
    console.log(`// From ${rinfo.address}:${rinfo.port}  (${msg.length} bytes)`);
    console.log(formatHex(msg));
    console.log(""); // blank line between packets
  });

  sock.on("error", (err) => {
    console.error(`Capture error: ${err.message}`);
    sock.close(() => process.exit(1));
  });

  sock.on("listening", () => {
    const addr = sock.address();
    console.error(`Listening on UDP ${addr.address}:${addr.port} ...`);
    console.error("Press Ctrl-C to stop.");
  });

  process.on("SIGINT", () => {
    console.error("\nStopping capture; closing socket.");
    sock.close(() => process.exit(0));
  });

  sock.bind(port);
}

if (require.main === module) main();

module.exports = { main, formatHex };
