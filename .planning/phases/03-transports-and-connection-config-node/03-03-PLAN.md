    ```javascript
    class MqttTransport extends BaseTransport {
      constructor(config) {
        super(config);
        // config: { brokerUrl, qos, topicPrefix, username, password, reconnectPeriod, publisherId }
        this._client = null;
        this._protocolFallbackDone = false;
      }

      async connect() {
        if (this._client) return;
        return this._tryConnect(5);
      }

      _tryConnect(protocolVersion) {
        return new Promise((resolve, reject) => {
          const opts = {
            protocolVersion,
            reconnectPeriod: (this.config.reconnectPeriod != null) ? this.config.reconnectPeriod : 5000,
            connectTimeout: 30000,
            clean: true,
            // T-03-02: NEVER pass rejectUnauthorized — let Node's default TLS validation apply
            // (mqtts:// scheme triggers TLS via the mqtt library; user-controlled bypass is forbidden)
          };
          if (this.config.username) opts.username = this.config.username;
          if (this.config.password) opts.password = this.config.password;

          const client = mqtt.connect(this.config.brokerUrl, opts);
          this._client = client;

          client.on("connect", () => {
            this._protocolFallbackDone = true;
            this.emit("connected");
            resolve();
          });

          client.on("error", (err) => {
            // T-03-03 redaction handled by Connection-Node's logger (not in this layer);
            // here we relay raw error to whoever listens.
            if (!this._protocolFallbackDone && protocolVersion === 5 && /unsupported protocol|unacceptable protocol version|protocol version not supported/i.test(err && err.message)) {
              // 5.0 → 3.1.1 fallback per Pitfall 3
              this._protocolFallbackDone = true;   // prevent third attempt
              try { client.end(true, {}, () => {}); } catch (e) { /* ignore */ }
              this._client = null;
              this._tryConnect(4).then(resolve).catch(reject);
              return;
            }
            this.emit("error", err);
            // Do NOT reject the connect Promise after first 'connect' fired — error events post-connect
            // are non-fatal (library will reconnect via reconnectPeriod).
          });

          client.on("close",     () => this.emit("disconnected"));
          client.on("reconnect", () => this.emit("reconnecting"));
          client.on("message", (topic, payload, packet) => {
            this.emit("message", payload, { topic, packet });
          });
        });
      }

      async close() {
        return new Promise((resolve) => {
          const client = this._client;
          this._client = null;
          this._protocolFallbackDone = false;   // W-3 fix: reset so a subsequent connect() can fall back again
          if (!client) return resolve();
          // Pitfall 4: client.end callback receives no arguments
          client.end(false, {}, () => resolve());
        });
      }

      send(payload, opts) {
        // (Implemented in Task 2)
        throw new Error("Task 2 implements send()");
      }
    }
    ```

    ```javascript
    // Topic-injection guard helper — module-level constant
    const TOPIC_FORBIDDEN = /[\/+#\x00-\x1F\x7F]/;

    function _validateTopicComponent(name, value) {
      if (value === undefined || value === null || String(value).length === 0) {
        throw createError(`TOPIC_INVALID_CHARACTER: ${name} is empty or undefined`);
      }
      if (TOPIC_FORBIDDEN.test(String(value))) {
        throw createError(`TOPIC_INVALID_CHARACTER: ${name} contains forbidden character (one of / + # or control char)`);
      }
    }

    // Inside MqttTransport class:
    _buildTopic(opts) {
      const prefix = this.config.topicPrefix || "ua";
      const pubId  = this.config.publisherId;
      const wgId   = opts && opts.writerGroupId;
      const dswId  = opts && opts.dataSetWriterId;

      // T-03-04: validate every user-controlled component before concat
      _validateTopicComponent("topicPrefix",     prefix);
      _validateTopicComponent("publisherId",     pubId);
      _validateTopicComponent("writerGroupId",   wgId);
      _validateTopicComponent("dataSetWriterId", dswId);

      return `${prefix}/${pubId}/${wgId}/${dswId}`;
    }

    send(payload, opts) {
      if (!this._client) {
        this.emit("error", createError("MQTT_SEND_NOT_CONNECTED: client is null"));
        return;
      }
      let topic;
      try {
        topic = this._buildTopic(opts);
      } catch (err) {
        // Topic injection / missing identifier — surface synchronously via throw so caller
        // (Plan 03-04 Connection-Node + Phase 4 Publisher) cannot accidentally publish to
        // a malformed/wildcarded topic.
        throw err;
      }

      const qos = (opts && opts.qos != null) ? opts.qos : (this.config.qos != null ? this.config.qos : 1);
      const publishOpts = {
        qos,
        retain: false,    // T-03-06: HARDCODED, NEVER caller-overridable (Part 14 §7.3.4 mandate)
      };

      const chunks = Array.isArray(payload) ? payload : [payload];
      for (const chunk of chunks) {
        this._client.publish(topic, chunk, publishOpts, (err) => {
          if (err) this.emit("error", createError(`MQTT_PUBLISH_ERROR: ${err.message}`));
        });
      }
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Network → MqttTransport | Untrusted MQTT messages enter via mqtt.js client; payload bytes are caller-trusted (encoder), but topic + packet metadata may be attacker-shaped |
| User config → topic concatenation | `publisherId`, `topicPrefix`, `writerGroupId`, `dataSetWriterId` are user input from the editor or msg fields — direct string concat is a classic injection vector |
| Caller send() opts → publish behavior | Callers (Phase 4 Publisher) might attempt to set `retain: true` or `rejectUnauthorized: false` — both must be structurally impossible, not just "discouraged" |
| TLS scheme parsing | `mqtts://` triggers TLS via mqtt.js library; user-controlled bypass would defeat encryption |
