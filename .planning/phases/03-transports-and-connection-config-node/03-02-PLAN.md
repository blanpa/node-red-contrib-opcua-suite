    ```javascript
    class UdpTransport extends BaseTransport {
      constructor(config) {
        super(config);
        // config: { port, multicastGroup, multicastInterface, mtu }
        this._socket  = null;
        this._chunks  = new Map();   // (Task 2 will populate this)
      }

      async connect() {
        if (this._socket) return;     // idempotent guard
        return new Promise((resolve, reject) => {
          this._socket = dgram.createSocket({ type: "udp4", reuseAddr: true });
          this._socket.on("error",   (err) => this.emit("error", createError(`UDP_SOCKET_ERROR: ${err.message}`)));
          this._socket.on("message", (buf, rinfo) => this._onDatagram(buf, rinfo));   // (Task 2 implements _onDatagram)
          this._socket.bind({ port: this.config.port, address: "0.0.0.0" }, () => {
            try {
              this._socket.addMembership(this.config.multicastGroup, this.config.multicastInterface || "0.0.0.0");
              this._socket.setMulticastLoopback(true);
              this._socket.setMulticastTTL(128);
              if (this.config.multicastInterface && this.config.multicastInterface !== "0.0.0.0") {
                this._socket.setMulticastInterface(this.config.multicastInterface);
              }
              this.emit("connected");
              resolve();
            } catch (err) {
              reject(createError(`UDP_BIND_FAILED: ${err.message}`));
            }
          });
        });
      }

      async close() {
        return new Promise((resolve) => {
          if (!this._socket) return resolve();
          const sock = this._socket;
          this._socket = null;
          sock.close(() => { this.emit("disconnected"); resolve(); });
        });
      }

      send(payload, opts) {
        if (!this._socket) {
          this.emit("error", createError("UDP_SEND_NOT_CONNECTED: socket is null"));
          return;
        }
        const chunks = Array.isArray(payload) ? payload : [payload];
        for (const chunk of chunks) {
          this._socket.send(chunk, this.config.port, this.config.multicastGroup, (err) => {
            if (err) this.emit("error", createError(`UDP_SEND_ERROR: ${err.message}`));
          });
        }
      }

      _onDatagram(buf, rinfo) {
        // Task 2 — chunk reassembly placeholder. For Task 1, simplest impl:
        try {
          this.emit("message", buf, { rinfo });
        } catch (e) { /* never throw out of dgram listener */ }
      }
    }
    module.exports = { UdpTransport };
    ```

    ```javascript
    _onDatagram(buf, rinfo) {
      let partial;
      try {
        partial = decodeNetworkMessage(buf);
      } catch (err) {
        // T-03-01: malformed datagram — log and continue
        this.emit("error", createError(`UDP_DECODE_ERROR: ${err.message}`));
        return;
      }

      // Single-buffer message — pass through directly
      if (!partial || !partial.chunk) {
        this.emit("message", buf, { rinfo });
        return;
      }

      // Chunked message — accumulate
      // VERIFIED FIELD PATHS (lib/uadp-encoder.js decodeNetworkMessage line 928+):
      //   - partial.publisherId               (top-level)
      //   - partial.groupHeader.writerGroupId (under groupHeader, line 964-965)
      //   - partial.chunk.messageSequenceNumber (under chunk, line 996)
      // The chunk header carries messageSequenceNumber NOT sequenceNumber; the
      // groupHeader's sequenceNumber is the GROUP sequence, not the per-message id.
      const key = `${partial.publisherId}|${partial.groupHeader.writerGroupId}|${partial.chunk.messageSequenceNumber}`;

      // Sweep expired (30s expiry) — runs every receive
      const now = Date.now();
      for (const [k, v] of this._chunks) {
        if (v.expiresAt < now) this._chunks.delete(k);
      }

      // Overflow guard (T-03-05): bound to 1000 in-flight reassemblies, drop-oldest
      if (!this._chunks.has(key) && this._chunks.size >= 1000) {
        const oldestKey = this._chunks.keys().next().value;   // Map insertion order
        this._chunks.delete(oldestKey);
        this.emit("warn", createError(`UDP_REASSEMBLY_OVERFLOW: dropped oldest key ${oldestKey}`));
      }

      // Add this chunk
      let entry = this._chunks.get(key);
      if (!entry) {
        entry = { totalSize: partial.chunk.totalSize, parts: new Map(), expiresAt: now + 30000 };
        this._chunks.set(key, entry);
      }
      entry.parts.set(partial.chunk.chunkOffset, partial.chunk.chunkData);

      // Check completeness
      let assembled = 0;
      for (const data of entry.parts.values()) assembled += data.length;
      if (assembled < entry.totalSize) return;   // incomplete — wait for more chunks

      // Complete — concat in offset order
      const sorted = [...entry.parts.entries()].sort((a, b) => a[0] - b[0]);
      const complete = Buffer.concat(sorted.map(([, d]) => d));
      this._chunks.delete(key);
      this.emit("message", complete, { rinfo });
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
